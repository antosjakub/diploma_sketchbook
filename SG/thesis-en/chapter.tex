\subsection{The core setup}

Consider a PDE defined on a d-dimensional unit cube
$$
\Omega := (0,1)^d,
$$


As the basic building block, we take standard linear hat functions

$$
\phi(x)=
\begin{cases}
1-|x|, & |x|\le 1,\\
0, & |x|>1.
\end{cases}
$$

We then create a set of equidistant grids $\Omega_l$ of level $l\in\mathbb{N}$ on the one dimensional unit interval $\bar{\Omega}=\left[0,1\right]$.

$$
\Omega_l := \{x_{l,i}=i\cdot h_l : 0\le i\le 2^l\}
$$

Where, for each level $l$, we define the mesh width
$$
h_l := 2^{-l}
$$

The standard hat funciton $\phi$ is then taken to generate a set of basis functions $\phi_{l,i}$ with the support $\left[ x_{l,i}-h_l, x_{l,i}+h_l \right]$.


$$
\phi_{l,i}(x)
=
\phi\!\left(\frac{x-i h_l}{h_l}\right),
\qquad 1\le i\le 2^l-1.
$$
The corresponding finite element space is
$$
V_l := \mathrm{span}\{\phi_{l,i} : 1\le i\le 2^l-1\}
$$


Instead of working with the nested spaces
$$
V_1 \subset V_2 \subset \cdots,
$$
we decompose them into intertwining increments. Define the hierarchical index sets
$$
I_l := \{\, i\in\mathbb{N} : 1\le i\le 2^l-1,\ i\ \text{odd}\,\}
$$
The \textit{hierarchical increment spaces} are then given as
$$
W_l := \mathrm{span}\{\phi_{l,i}: i\in I_l\}.
$$

Then, comparing the definition of $V_l$ and $W_l$, one gets the direct-sum decomposition
$$
V_l = \bigoplus_{k=1}^l W_k.
$$


Every $u\in V_l$ then admites a unique hierarchical representation
$$
u(x)=\sum_{k=1}^l \sum_{i\in I_k} v_{k,i}\,\phi_{k,i}(x),
$$
where $v_{k,i}$ are called \textit{hierarchical surpluses}.

\subsection{Algorithm}

\begin{algorithm}[H]
\caption{Mini-batch PINN training}
\begin{algorithmic}[1]
    \Require Neural network $u_\theta: \mathbb{R}^{d+1}\to\mathbb{R}$, sampled data $\mathbf{X}_{pde}, \mathbf{X}_{ic}, \mathbf{X}_{bc}$
    \State \textbf{Initialize} network parameters $\theta$
    \For{epoch $= 1, \dots, E$}
        \State Shuffle training data
        \For{each mini-batch $\mathcal{B} \subset \{1,\dots,N\}$, $|\mathcal{B}|=B$}
            \State \textbf{Forward pass:} compute $\hat{y}_i = \mathcal{N}_\theta(x_i)$ for $i\in\mathcal{B}$
            \State \textbf{Compute loss:}
            $\mathcal{L}(\theta) = \dfrac{1}{B}\displaystyle\sum_{i\in\mathcal{B}}
                \bigl(\hat{y}_i - y_i\bigr)^2$
            \State \textbf{Backward pass:} compute $\nabla_\theta \mathcal{L}$ via backpropagation
            \State \textbf{Update:} $\theta \leftarrow \theta - \eta\,\nabla_\theta \mathcal{L}$
        \EndFor
        \State Every $k$ epochs, regenerate training data
    \EndFor
    \State \Return $\theta$
\end{algorithmic}
\end{algorithm}
