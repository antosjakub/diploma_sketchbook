\frametitle{Problem statement}

\pause
Let
\begin{itemize}
\item $\Omega \subset \mathbb{R}^d$ be open and bounded
\item $T \in (0, \infty)$
%\item $Q_T = (0,T) \times \Omega$
\end{itemize}

\pause
Consider
\begin{align*}
\partial_t u + L\,u = f \quad &\text{in } (0,T)\times\Omega \\
u(0,x) = u_{IC}(x)\quad &\text{on } \Omega \\
u(t,x) = u_{BC}(t,x)\quad &\text{on } (0,T)\times\partial\Omega
\end{align*}

\pause
Where
\begin{itemize}
\item $L\,u = -\alpha\,\Delta u + \vec{b} \cdot \nabla u + c\, u$
\pause
\item $u,f,c,u_{BC}: Q_T \to \mathbb{R}$
\item $u_{IC}: \Omega \to \mathbb{R}$
\item $b: Q_T \to \mathbb{R}^d$
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Problem statement}

Now take
$$ \vec{b}=\vec{0}, \quad c = 0, \quad u_{BC} = 0,\quad \Omega = (0,1)^d $$
\pause
and consider the system
\begin{align*}
\partial_t u - \alpha\,\Delta u = f \quad &\text{in } (0,t)\times\Omega \\
u(0,x) = u_{IC}(x)\quad &\text{on } \Omega \\
u(t,x) = 0\quad &\text{on } (0,T)\times\partial\Omega
\end{align*}

\pause
Moreover
$$
f(t,x) = -\Bigl(\prod_{i=1}^d \sin{(\pi k_i x_i)}\Bigr) 
\Bigl(a\sin{(at)} + (\beta - \alpha\,\pi^2\,|k|^2)\cos{(at)}\Bigr) e^{-\beta\,t}
$$
$$ u_{analytic}(t,x) = \Bigl(\prod_{i=1}^d \sin{(\pi k_i x_i)}\Bigr) \cos{(at)}\,e^{-\beta t} $$

\end{frame}
%%%%%%%%%%%%%%%% End

%%%%%%%%%%%%%%%% Begin
\begin{frame}
\frametitle{FDM (Finite difference method, konecne diference)}

$$ \partial_t u - \alpha\,\Delta u = f $$

\pause
Finite differences
\begin{itemize}

\item Forward difference in time

$\partial u / \partial t \approx \frac{u(t+\tau,x) - u(t,x)}{\tau} $

\item Central difference in space

$\partial u / \partial x \approx \frac{u(t,x+h/2) - u(t,x-h/2)}{h} $

$\partial^2 u /\partial x^2 \approx \frac{u(t,x+h) - 2u(t,x) + u(t,x-h)}{h^2} $

\end{itemize}

\pause
Create a uniform grid in time and space

$$ \{t_j\}_{j=0}^{n_t} = (t_0, t_1, ..., t_n, t_{n_t}), \quad t_0 = 0.0, \quad t_{n_t} = T $$
$$ \{x_i\}_{i=0}^{n+1} = (x_0, x_1, ..., x_n, x_{n+1}), \quad x_0 = 0.0, \quad x_{n+1} = 1.0 $$

\pause
Define
$$ u^j_i = u(t_j, x_i) $$



\end{frame}
%%%%%%%%%%%%%%%% End

%%%%%%%%%%%%%%%% Begin
\begin{frame}
\frametitle{FDM: Towards the $A\,x=b$}

implicit > explicit (explicit needs $\alpha \tau / h^2 \le 1/2$)

\pause
$$
\frac{u_i^{t+1} - u_i^{t}}{\tau}
- \alpha\, \frac{u_{i+1}^{t+1} - 2u_i^{t+1} + u_{i-1}^{t+1}}{h^2} = f_i^{t+1}
$$

$$
u_i^{t+1} -
\frac{\alpha\,\tau}{h^2} (u_{i+1}^{t+1} - 2u_i^{t+1} + u_{i-1}^{t+1})
= u_i^{t} + \tau f_i^{t+1}
$$

\pause
$$
U^{t+1} -
\frac{\alpha\,\tau}{h^2} L\,U^{t+1}
= (I - \frac{\alpha\,\tau}{h^2} L)\, U^{t+1}
= U^t + \tau F^{t+1}
$$

\pause
To get $U^{t+1}$ from $U^{t}$, we need to solve:

$$
(I - \frac{\alpha\,\tau}{h^2} L)\, U^{t+1}
= U^t + \tau F^{t+1}
$$

\pause
We need to solve $A\,x=b$, where $x$ and $b$ are of size $N=n^d$.
\begin{itemize}
\item $A = (I - \frac{\alpha\,\tau}{h^2} L)$
\item $x = U^{t+1}$
\item $b = U^t + \tau F^{t+1}$
\end{itemize}

\end{frame}
%%%%%%%%%%%%%%%% End

\frametitle{FDM: The Curse of Dimensionality}

\textbf{Answer: Yes!}
\pause
\begin{itemize}
    \item Conjugate gradients (CG)
    \begin{itemize}
        \item solves Ax=b, for x where A is a symmetric, positive-definite linear operator
        \item uses short recurrences $\implies$ fixed memory costs and fixed computational costs per iteration
    \end{itemize}
\end{itemize}

\pause
\textbf{Answer: Not really...}
\pause
\begin{itemize}
    \item Problem = memory
\end{itemize}

\pause
\begin{lstlisting}[breaklines=true,frame=single,basicstyle=\footnotesize\ttfamily]
X = get_grid_as_1d_vector(n,d); # N x d
U = u_IC.(X, t=0.0)             # N
L = get_laplace_matrix()        # N x (2d+1)
I = sparse()                    # N
A = I - r * L                   # N x (2d+1)
                        # total = N x (5d+4)
for loop
    F = f.(X, t+tau)            # N
    U = cg(A, U + tau * F)      # N x (3+1)
\end{lstlisting}

\frametitle{NN training}

\begin{algorithm}[H]
\caption{Standard NN Training (Mini-Batch Gradient Descent)}
\begin{algorithmic}[1]
    \Require $\mathcal{N}:\mathbb{R}^n\to\mathbb{R}^m$, training data $\{(x_i, y_i)\}_{i=1}^{N}$
    \State \textbf{Initialize} network parameters $\theta$
    \For{epoch $= 1, \dots, E$}
        \State Shuffle training data
        \For{each mini-batch $\mathcal{B} \subset \{1,\dots,N\}$, $|\mathcal{B}|=B$}
            \State \textbf{Forward pass:} compute $\hat{y}_i = \mathcal{N}_\theta(x_i)$ for $i\in\mathcal{B}$
            \State \textbf{Compute loss:}
            $\mathcal{L}(\theta) = \dfrac{1}{B}\displaystyle\sum_{i\in\mathcal{B}}
                \bigl(\hat{y}_i - y_i\bigr)^2$
            \State \textbf{Backward pass:} compute $\nabla_\theta \mathcal{L}$ via backpropagation
            \State \textbf{Update:} $\theta \leftarrow \theta - \eta\,\nabla_\theta \mathcal{L}$
        \EndFor
    \EndFor
    \State \Return $\theta$
\end{algorithmic}
\end{algorithm}

\end{frame}

\begin{algorithm}[H]
\caption{PINN Training}
\begin{algorithmic}[1]
    \Require Neural network $u_\theta: \mathbb{R}^{d+1}\to\mathbb{R}$, weights $\lambda_{pde}, \lambda_{ic}, \lambda_{bc}$
    \State \textbf{Initialize} network parameters $\theta$
    \For{step $= 1, \dots, S$}
        \State \textbf{Sample the 3 domains}
        \State \textbf{Forward pass}
                %$\hat{u}^i_{pde} = u_\theta(x_i^d)$
                %$\hat{u}_i = \mathcal{N}_\theta(x_i^d)$
        \State \textbf{Compute loss for each redisual:}
               \State $\mathcal{L}_{pde} = \mathbb{E}\left[(\partial_t\,u_{\theta} - \alpha\,\Delta\,u_{\theta} - f)^2\right]_{\sim\:\mathcal{U}((0,T)\times\Omega)}$
               \State $\mathcal{L}_{ic} = \mathbb{E}\left[(u_{\theta} - u_{ic})^2\right]_{\sim\:\mathcal{U}(\Omega)}$
               \State $\mathcal{L}_{bc} = \mathbb{E}\left[(u_{\theta} - u_{bc})^2\right]_{\sim\:\mathcal{U}((0,T)\times\partial\Omega)}$
               %\State $\mathcal{L}_{pde} = \mathbb{E}\left[(\partial_t\,\hat{u}_{pde} - \alpha\,\Delta\,\hat{u}_{pde} - f)^2\right]_{\sim\:\mathcal{U}((0,T)\times\Omega)}$
               %\State $\mathcal{L}_{ic} = \mathbb{E}\left[(\hat{u}_{ic} - u_{ic})^2\right]_{\sim\:\mathcal{U}(\Omega)}$
               %\State $\mathcal{L}_{bc} = \mathbb{E}\left[(\hat{u}_{bc} - u_{bc})^2\right]_{\sim\:\mathcal{U}((0,T)\times\partial\Omega)}$

               %$\mathcal{R}^{pde} = \partial_t\,\mathcal{N}_\theta(X) - \alpha\,\Delta\,\mathcal{N}_\theta(X) - f$ \\
        %\State $\mathcal{R}_{pde} = \partial_t\,\hat{u}_{pde} - \alpha\,\Delta\,\hat{u}_{pde} - f(X)$
        %\State $\mathcal{R}_{bc} = \hat{u}_{pde} - u_{bc}(X)$
        %\State $\mathcal{R}_{ic} = \hat{u}_{ic} - u_{ic}(X)$
        %       $\mathcal{R} = \mathcal{F}\!\left[\mathcal{N}_\theta\right](x_j^c)$
        %\State \textbf{PDE loss:}
        %       $\mathcal{L}_{\text{data}} = \dfrac{1}{N_d}\displaystyle\sum_i \bigl(\hat{u}_i - u_i^d\bigr)^2$
        %\State \textbf{BC loss:}
        %       $\mathcal{L}_{\text{data}} = \dfrac{1}{N_d}\displaystyle\sum_i \bigl(\hat{u}_i - u_i^d\bigr)^2$
        %\State \textbf{IC loss:}
        %       $\mathcal{L}_{\text{phys}} = \dfrac{1}{N_c}\displaystyle\sum_j r_j^2$
        \State \textbf{Total loss:}
               $\mathcal{L} = \lambda_{pde}\mathcal{L}_{pde} + \lambda_{ic}\,\mathcal{L}_{ic} + \lambda_{bc}\,\mathcal{L}_{bc}$
        \State \textbf{Backward pass \& update:}
               $\theta \leftarrow \theta - \eta\,\nabla_\theta \mathcal{L}$
    \EndFor
    \State \Return $\theta$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\caption{PINN Training}
\begin{algorithmic}[1]

                $\{(x_{bc}^i,t_{bc}^i)\}\sim\mathcal{U}(Q_T)$
                %$X_{ic}\in\mathbb{R}^{N_{ic}\times d}$,
                %$X_{bc}\in\mathbb{R}^{N_{bc}\times(d+1)}$,
 
        \State \textbf{Forward pass:}
                $\hat{u}_i^{pde} = \mathcal{N}_\theta(x_i^d)$
                $\hat{u}_i = \mathcal{N}_\theta(x_i^d)$
        \State \textbf{Residuals:} (computed via automatic differentiation)
               %$\mathcal{R}^{pde} = \partial_t\,\mathcal{N}_\theta(X) - \alpha\,\Delta\,\mathcal{N}_\theta(X) - f$ \\
        \State $\mathcal{R}_{pde} = \partial_t\,\hat{u}_{pde} - \alpha\,\Delta\,\hat{u}_{pde} - f(X)$
        \State $\mathcal{R}_{bc} = \hat{u}_{pde} - u_{bc}(X)$
        \State $\mathcal{R}_{ic} = \hat{u}_{ic} - u_{ic}(X)$
        %       $\mathcal{R} = \mathcal{F}\!\left[\mathcal{N}_\theta\right](x_j^c)$
        %\State \textbf{PDE loss:}
        %       $\mathcal{L}_{\text{data}} = \dfrac{1}{N_d}\displaystyle\sum_i \bigl(\hat{u}_i - u_i^d\bigr)^2$
        %\State \textbf{BC loss:}
        %       $\mathcal{L}_{\text{data}} = \dfrac{1}{N_d}\displaystyle\sum_i \bigl(\hat{u}_i - u_i^d\bigr)^2$
